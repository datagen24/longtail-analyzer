107    def update_activity(self, events: list[dict[str, Any]]) -> None:
108        """
109        Update profile with new activity events.
110
111        Args:
112            events: List of new events to incorporate
113        """
114        if not events:
115            return
116
117        # Update temporal tracking
118        event_times = [
119            datetime.fromisoformat(e.get("@timestamp", ""))
120            for e in events
121            if "@timestamp" in e
122        ]
123        if event_times:
124            self.last_seen = max(self.last_seen, max(event_times))
125            if not hasattr(self, "first_seen") or self.first_seen is None:
126                self.first_seen = min(event_times)
127            else:
128                self.first_seen = min(self.first_seen, min(event_times))
129
130        # Update activity counts
131        self.total_events += len(events)
132
133        # Extract unique targets
134        targets = set()
135        for event in events:
136            if "destination.ip" in event:
137                targets.add(event["destination.ip"])
138            if "destination.port" in event:
139                targets.add(str(event["destination.port"]))
140
141        self.unique_targets = len(targets)
142
143        # Update last analysis time
144        self.last_analysis = datetime.now()
