 93    async def _call_ollama(self, prompt: str) -> str | None:
 94        """
 95        Call the Ollama API with a prompt.
 96
 97        Args:
 98            prompt: Prompt to send to the model
 99
100        Returns:
101            Model response or None if error
102        """
103        try:
104            request_data = {
105                "model": self.model,
106                "prompt": prompt,
107                "stream": False,
108                "options": {
109                    "temperature": self.temperature,
110                    "num_predict": self.max_tokens,
111                },
112            }
113
114            response = await self.session.post(
115                f"{self.base_url}/api/generate", json=request_data
116            )
117            response.raise_for_status()
118
119            result = response.json()
120            return result.get("response")
121
122        except httpx.TimeoutException:
123            logger.error("Timeout calling Ollama API")
124            return None
125        except httpx.HTTPStatusError as e:
126            logger.error(f"HTTP error calling Ollama API: {e}")
127            return None
128        except Exception as e:
129            logger.error(f"Unexpected error calling Ollama API: {e}")
130            return None

177    def _summarize_events(self, events: list[dict[str, Any]]) -> str:
178        """
179        Create a summary of events for analysis.
180
181        Args:
182            events: Events to summarize
183
184        Returns:
185            Event summary string
186        """
187        if not events:
188            return "No events"
189
190        # Extract key information
191        source_ips = set()
192        dest_ips = set()
193        dest_ports = set()
194        actions = set()
195        timestamps = []
196
197        for event in events:
198            if "source.ip" in event:
199                source_ips.add(event["source.ip"])
200            if "destination.ip" in event:
201                dest_ips.add(event["destination.ip"])
202            if "destination.port" in event:
203                dest_ports.add(str(event["destination.port"]))
204            if "event.action" in event:
205                actions.add(event["event.action"])
206            if "@timestamp" in event:
207                timestamps.append(event["@timestamp"])
208
209        summary = f"""
210Total Events: {len(events)}
211Source IPs: {len(source_ips)} ({', '.join(list(source_ips)[:5])}{'...' if len(source_ips) > 5 else ''})
212Destination IPs: {len(dest_ips)} ({', '.join(list(dest_ips)[:5])}{'...' if len(dest_ips) > 5 else ''})
213Destination Ports: {len(dest_ports)} ({', '.join(list(dest_ports)[:10])}{'...' if len(dest_ports) > 10 else ''})
214Actions: {', '.join(list(actions)[:10])}{'...' if len(actions) > 10 else ''}
215Time Range: {min(timestamps) if timestamps else 'unknown'} to {max(timestamps) if timestamps else 'unknown'}
216"""
217
218        return summary

220    def _parse_analysis_response(self, response: str) -> dict[str, Any]:
221        """
222        Parse the LLM response into structured data.
223
224        Args:
225            response: Raw LLM response
226
227        Returns:
228            Parsed analysis data
229        """
230        try:
231            # Try to extract JSON from the response
232            start_idx = response.find("{")
233            end_idx = response.rfind("}") + 1
234
235            if start_idx != -1 and end_idx > start_idx:
236                json_str = response[start_idx:end_idx]
237                analysis = json.loads(json_str)
238
239                # Validate and set defaults
240                analysis.setdefault("threat_score", 0.0)
241                analysis.setdefault("threat_level", "low")
242                analysis.setdefault("attack_patterns", [])
243                analysis.setdefault("ttps", [])
244                analysis.setdefault("confidence", 0.5)
245                analysis.setdefault("analysis", response)
246                analysis.setdefault("recommendations", [])
247                analysis.setdefault("needs_deep_analysis", False)
248
249                return analysis
250            else:
251                # Fallback if no JSON found
252                return {
253                    "threat_score": 0.5,
254                    "threat_level": "medium",
255                    "attack_patterns": [],
256                    "ttps": [],
257                    "confidence": 0.3,
258                    "analysis": response,
259                    "recommendations": [],
260                    "needs_deep_analysis": True,
261                }
262
263        except json.JSONDecodeError as e:
264            logger.error(f"Error parsing LLM response as JSON: {e}")
265            return {
266                "threat_score": 0.5,
267                "threat_level": "medium",
268                "attack_patterns": [],
269                "ttps": [],
270                "confidence": 0.3,
271                "analysis": response,
272                "recommendations": [],
273                "needs_deep_analysis": True,
274            }

502    def _summarize_events(self, events: list[dict[str, Any]]) -> str:
503        """
504        Create a summary of events for analysis.
505
506        Args:
507            events: Events to summarize
508
509        Returns:
510            Event summary string
511        """
512        if not events:

514
515        # Extract key information

518        dest_ports = set()

531            if "@timestamp" in event:

533
534        summary = f"""
535Total Events: {len(events)}
536Source IPs: {len(source_ips)} ({', '.join(list(source_ips)[:5])}{'...' if len(source_ips) > 5 else ''})
537Destination IPs: {len(dest_ips)} ({', '.join(list(dest_ips)[:5])}{'...' if len(dest_ips) > 5 else ''})
538Destination Ports: {len(dest_ports)} ({', '.join(list(dest_ports)[:10])}{'...' if len(dest_ports) > 10 else ''})
539Actions: {', '.join(list(actions)[:10])}{'...' if len(actions) > 10 else ''}
540Time Range: {min(timestamps) if timestamps else 'unknown'} to {max(timestamps) if timestamps else 'unknown'}
541"""
542
543        return summary
