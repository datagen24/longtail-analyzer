122        self,
123        query_embedding: np.ndarray,
124        top_k: int = 10,
125        filter_metadata: Optional[Dict[str, Any]] = None
126    ) -> List[Tuple[str, float, Dict[str, Any]]]:
127        """
128        Search for similar patterns using vector similarity.
129        
130        Args:
131            query_embedding: Query vector embedding
132            top_k: Number of similar patterns to return
133            filter_metadata: Optional metadata filters
134            
135        Returns:
136            List of (pattern_id, similarity_score, metadata) tuples
137        """
138        try:
139            # Convert numpy array to list
140            query_list = query_embedding.tolist()
141            
142            # Perform similarity search
143            results = self.collection.query(
144                query_embeddings=[query_list],
145                n_results=top_k,
146                where=filter_metadata
147            )
148            
149            # Format results
150            similar_patterns = []
151            if results['ids'] and results['ids'][0]:
152                for i, pattern_id in enumerate(results['ids'][0]):
153                    similarity = 1.0 - results['distances'][0][i]  # Convert distance to similarity
154                    metadata = results['metadatas'][0][i] if results['metadatas'] else {}
155                    similar_patterns.append((pattern_id, similarity, metadata))
156            
157            logger.debug(f"Found {len(similar_patterns)} similar patterns")
158            return similar_patterns
159            
160        except Exception as e:
161            logger.error(f"Error searching similar patterns: {e}")
162            return []

 36        
 37        # Create database connection

 40

 44    def _init_db(self) -> None:
 45        """Initialize SQLite tables for state storage."""
 46        # Analysis sessions table
 47        self.conn.execute("""
 48            CREATE TABLE IF NOT EXISTS analysis_sessions (
 49                session_id TEXT PRIMARY KEY,
 50                session_type TEXT NOT NULL,
 51                start_time TIMESTAMP NOT NULL,
 52                end_time TIMESTAMP,
 53                status TEXT DEFAULT 'active',
 54                config_data TEXT,
 55                metadata TEXT
 56            )
 57        """)
 58        
 59        # Analysis checkpoints table
 60        self.conn.execute("""
 61            CREATE TABLE IF NOT EXISTS analysis_checkpoints (
 62                checkpoint_id TEXT PRIMARY KEY,
 63                session_id TEXT NOT NULL,
 64                checkpoint_time TIMESTAMP NOT NULL,
 65                checkpoint_data TEXT NOT NULL,
 66                progress_percentage REAL DEFAULT 0.0,
 67                FOREIGN KEY (session_id) REFERENCES analysis_sessions (session_id)
 68            )
 69        """)
 70        
 71        # System state table
 72        self.conn.execute("""
 73            CREATE TABLE IF NOT EXISTS system_state (
 74                state_key TEXT PRIMARY KEY,
 75                state_value TEXT NOT NULL,
 76                last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
 77                description TEXT
 78            )
 79        """)
 80        
 81        # Analysis results table
 82        self.conn.execute("""
 83            CREATE TABLE IF NOT EXISTS analysis_results (
 84                result_id TEXT PRIMARY KEY,
 85                session_id TEXT NOT NULL,
 86                result_type TEXT NOT NULL,
 87                result_data TEXT NOT NULL,
 88                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
 89                FOREIGN KEY (session_id) REFERENCES analysis_sessions (session_id)
 90            )
 91        """)
