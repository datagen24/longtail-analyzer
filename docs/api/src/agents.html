 57    def generate_time_windows(
 58        self,
 59        start_date: datetime,
 60        end_date: datetime
 61    ) -> List[ProcessingWindow]:
 62        """
 63        Generate overlapping time windows for analysis.
 64        Windows overlap to ensure patterns spanning boundaries are detected.
 65        """
 66        windows = []
 67        current = start_date
 68        
 69        while current < end_date:
 70            window_end = min(
 71                current + timedelta(hours=self.window_hours),
 72                end_date
 73            )
 74            
 75            # Calculate overlap period with previous window
 76            overlap_start = None
 77            overlap_end = None
 78            
 79            if windows:  # If there's a previous window
 80                overlap_start = current
 81                overlap_end = current + timedelta(hours=self.overlap_hours)
 82            
 83            window = ProcessingWindow(
 84                window_id=self._get_window_id(current, window_end),
 85                start_time=current,
 86                end_time=window_end,
 87                overlap_start=overlap_start,
 88                overlap_end=overlap_end
 89            )
 90            
 91            windows.append(window)
 92            
 93            # Move forward with overlap
 94            current += timedelta(hours=self.window_hours - self.overlap_hours)
 95        
 96        logger.info(f"Generated {len(windows)} time windows from {start_date} to {end_date}")
 97        return windows

100        """Generate unique ID for a time window"""

104    async def process_window(
105        self,
106        window: ProcessingWindow,
107        pattern_analyzer=None
108    ) -> Dict:
109        """
110        Process a single time window with intelligent data retrieval.
111        
112        Key optimization: Get summary first, identify anomalies, then fetch details only for anomalous entities.
113        """
114        
115        # Skip if already processed
116        if window.window_id in self.processed_windows:
117            logger.info(f"Window {window.window_id} already processed, skipping")
118            return {"status": "skipped", "window_id": window.window_id}
119        
120        logger.info(f"Processing window {window.start_time} to {window.end_time}")
121        
122        try:
123            # Step 1: Get aggregated summary for the window
124            summary = await self.mcp_client.get_aggregated_summary(
125                window.start_time,
126                window.end_time
127            )
128            window.summary = summary
129            
130            logger.info(f"Window summary: {summary.total_events} events, "
131                       f"{summary.unique_sources} unique sources")
132            
133            # Step 2: Identify anomalous entities from summary
134            anomalies = await self.query_optimizer.identify_anomalies(
135                summary,
136                threshold_multiplier=2.0
137            )
138            
139            # Limit to max entities per window
140            anomalies = anomalies[:self.max_entities_per_window]
141            
142            logger.info(f"Identified {len(anomalies)} anomalous entities for detailed analysis")
143            
144            # Step 3: Process overlap region if exists
145            overlap_entities = []
146            if window.overlap_start and window.overlap_end:
147                overlap_entities = await self._process_overlap(
148                    window.overlap_start,
149                    window.overlap_end,
150                    anomalies
151                )
152            
153            # Step 4: Get detailed data for anomalous entities
154            entity_events = await self.query_optimizer.fetch_anomaly_details(
155                anomalies,
156                time_range_hours=self.window_hours
157            )
158            
159            # Step 5: Analyze patterns if analyzer provided
160            patterns = {}
161            if pattern_analyzer and entity_events:
162                patterns = await pattern_analyzer.analyze(entity_events)
163                window.patterns_found = len(patterns)
164            
165            # Step 6: Update profiles incrementally
166            profiles_updated = 0
167            for entity_id, events in entity_events.items():
168                # Determine entity type
169                entity_type = self._determine_entity_type(entity_id)
170                
171                # Update profile
172                profile = self.profile_manager.update_profile_incrementally(
173                    entity_id=entity_id,
174                    entity_type=entity_type,
175                    new_events=events,
176                    llm_analysis=patterns.get(entity_id)
177                )
178                
179                profiles_updated += 1
180                
181                # Check for related entities
182                if profile.related_entities:
183                    await self._process_related_entities(profile.related_entities)
184            
185            # Step 7: Handle cross-boundary patterns
186            if overlap_entities:
187                await self._merge_cross_boundary_patterns(overlap_entities, entity_events)
188            
189            # Mark window as processed
190            window.processed = True
191            window.entities_found = len(entity_events)
192            self.processed_windows.add(window.window_id)
193            self.window_cache[window.window_id] = window
194            
195            # Save checkpoint
196            self._save_checkpoint(window)
197            
198            result = {
199                "status": "processed",
200                "window_id": window.window_id,
201                "total_events": summary.total_events,
202                "entities_analyzed": len(entity_events),
203                "profiles_updated": profiles_updated,
204                "patterns_found": window.patterns_found,
205                "anomalies_detected": len(anomalies)
206            }
207            
208            logger.info(f"Window processing complete: {result}")
209            return result
210            
211        except Exception as e:
212            logger.error(f"Error processing window {window.window_id}: {e}")
213            return {
214                "status": "error",
215                "window_id": window.window_id,
216                "error": str(e)
217            }

219    async def _process_overlap(
220        self,
221        overlap_start: datetime,
222        overlap_end: datetime,
223        current_anomalies: List[Dict]
224    ) -> List[str]:
225        """
226        Process overlap region between windows to detect cross-boundary patterns.
227        """
228        logger.info(f"Processing overlap region: {overlap_start} to {overlap_end}")
229        
230        # Get summary for overlap period
231        overlap_summary = await self.mcp_client.get_aggregated_summary(
232            overlap_start,
233            overlap_end
234        )
235        
236        # Find entities that appear in both windows
237        overlap_entities = []
238        current_entity_ids = [a["entity_id"] for a in current_anomalies]
239        
240        for source in overlap_summary.top_sources:
241            if source["ip"] in current_entity_ids:
242                overlap_entities.append(source["ip"])
243        
244        logger.info(f"Found {len(overlap_entities)} entities in overlap region")
245        return overlap_entities

247    async def _merge_cross_boundary_patterns(
248        self,
249        overlap_entities: List[str],
250        entity_events: Dict[str, List[Dict]]
251    ):
252        """
253        Merge patterns that span window boundaries.
254        """
255        for entity_id in overlap_entities:
256            if entity_id not in entity_events:
257                continue
258            
259            # Get the profile
260            profile = self.profile_manager.get_profile(entity_id)
261            if not profile:
262                continue
263            
264            # Check if patterns in current window connect with previous patterns
265            current_events = entity_events[entity_id]
266            
267            # Look for pattern continuation
268            for event in current_events:
269                event_time = datetime.fromisoformat(event.get("@timestamp", datetime.now().isoformat()))
270                
271                # Check if this event is close in time to existing patterns
272                for pattern in profile.scanning_patterns + profile.exploitation_attempts:
273                    time_gap = (event_time - pattern.last_seen).total_seconds() / 60
274                    
275                    if time_gap < 30:  # Within 30 minutes
276                        # This is likely a continuation of the pattern
277                        pattern.last_seen = event_time
278                        pattern.occurrence_count += 1
279                        logger.info(f"Merged cross-boundary pattern for {entity_id}")
280            
281            # Save updated profile
282            self.profile_manager.save_profile(profile)

309    def _determine_entity_type(self, entity_id: str) -> EntityType:
310        """Determine the type of entity from its ID"""
311        if entity_id.startswith("ip_"):
312            return EntityType.IP
313        elif entity_id.startswith("asn_") or entity_id.startswith("AS"):
314            return EntityType.ASN
315        elif entity_id.startswith("port_"):
316            return EntityType.PORT
317        elif entity_id.startswith("country_"):
318            return EntityType.COUNTRY
319        else:
320            # Try to infer from format
321            if "." in entity_id and all(
322                part.isdigit() and 0 <= int(part) <= 255
323                for part in entity_id.split(".")
324            ):
325                return EntityType.IP
326            return EntityType.COMPOSITE

363        # Load previous state if resuming

366        
367        # Process statistics

376        for i, window in enumerate(windows):

381                skipped += 1
382                continue
383            
384            # Process window
385            result = await self.process_window(window, pattern_analyzer)
386            
387            if result["status"] == "processed":
388                processed += 1
389                total_entities += result.get("entities_analyzed", 0)
390                total_patterns += result.get("patterns_found", 0)
391            elif result["status"] == "error":
392                errors += 1
393            
394            # Adaptive delay based on data volume
395            if window.summary and window.summary.total_events > 10000:
396                await asyncio.sleep(2)  # Longer delay for heavy windows
397            else:
398                await asyncio.sleep(0.5)
399        
400        return {
401            "start_date": start_date.isoformat(),
402            "end_date": end_date.isoformat(),
403            "total_windows": total_windows,
404            "processed": processed,
405            "skipped": skipped,
406            "errors": errors,
407            "total_entities_analyzed": total_entities,
408            "total_patterns_found": total_patterns
409        }
